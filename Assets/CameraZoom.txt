using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using System;

public class CameraZoom : MonoBehaviour
{
    public int zoom = 20;
    public int zoomSprint = 100;
    public int normal = 60;
    public float smooth = 5;

    public Camera Camcorder;
    public AudioSource Zin;

    public PlayerInput playerInput;

    private bool isZoomed = false;

    // Update is called once per frame
    void Update()
    {
        if(Input.GetKeyDown(KeyCode.Q))
        {
            isZoomed = !isZoomed;
        }

        if(isZoomed)
        {
            Zin.Play();
            Camcorder.fieldOfView = Mathf.Lerp(Camcorder.fieldOfView, zoom, Time.deltaTime * smooth);
            

        } else if(!isZoomed)
        {
            Zin.Stop();
            Camcorder.fieldOfView = Mathf.Lerp(Camcorder.fieldOfView, normal, Time.deltaTime * smooth);
        }

        if(playerInput.isSprint)
        {
            Camcorder.fieldOfView = Mathf.Lerp(Camcorder.fieldOfView, zoomSprint, Time.deltaTime * smooth);
        } else if(!playerInput.isSprint)
        {
            Camcorder.fieldOfView = Mathf.Lerp(Camcorder.fieldOfView, normal, Time.deltaTime * smooth);
        }
    }
}
